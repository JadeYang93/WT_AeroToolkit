# 气动组工具箱 - Python 依赖
#
# 安装方式（内网用户）：
#   双击「一键部署.bat」
# 或手动：
#   pip install -r requirements.txt -i http://sanynexus.sany.com.cn/repository/pypi-aliyun/simple/ --trusted-host sanynexus.sany.com.cn
#
# Python 版本要求：3.8+（PyQt5 5.15 LTS、scipy PCHIP 都需要）

PyQt5 >= 5.15          # GUI 框架（5.15 LTS 支持 HighDpiScaleFactorRoundingPolicy）
numpy >= 1.20          # 数值计算
pandas >= 1.3          # Excel / CSV 数据处理
scipy >= 1.6           # PCHIP 插值（shape_design 后缘修正）
openpyxl >= 3.0        # Excel 读写（read_only 模式需要 wb.close()）
matplotlib >= 3.4      # 绘图（TE 对比图、参数分布图）
psutil >= 5.8          # FOCUS6 进程管理（超时强杀）
